\subsection{Command Line Arguments}

\begin{frame}[fragile]
\frametitle{Command Line Arguments}
\begin{lstlisting}
int main(int argc, char **argv) { ... }
\end{lstlisting}
argc: Argument Count\\
argv: Argument Values / Vector
\vspace{3mm}
{\tiny
\begin{lstlisting}
int main(int argc, char **argv) {
  cout << argc << endl;
  for (int i=0; i<argc; i++) {
	cout << argv[i] << endl;
  }
  return 0;
}
\end{lstlisting}
\verb|a.out v1 v2 "v3 v4"|\\
\vspace{3mm}
a.out\\
v1\\
v2\\
v3 v4\\
}
\end{frame}
